#### 1: 下面的代码输出多少？修改代码让 fnArr[i]() 输出 i。使用 两种以上的方法
```
var fnArr = [];
for (var i = 0; i < 10; i ++) {
    fnArr[i] =  function(){
        return i;
    };
}
console.log( fnArr[3]() );  //
```
- 输出的结果为10
- 第一种方法
    ```
    var fnArr = [];
    for (var i = 0; i < 10; i ++) {
        fnArr[i] =  function(i){
            return function (){
                return i
            }        
        }(i)
    }
    console.log( fnArr[3]() );  //3
    ```
- 第二种方法
    ```
    var fnArr = [];
    for (var i = 0; i < 10; i ++) {
        ~function (i){
            fnArr[i] = function(){
                return i
            }
        }(i)
    }
    console.log( fnArr[3]() );  //3
    ```
